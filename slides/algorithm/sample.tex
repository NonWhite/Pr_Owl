\begin{lstlisting}
	GreedySearch( Dataset |$D$| , Solution |$G_0$| ) : return a BN |$G$|
		|$G = G_0$|
		For a number of iterations |$K$|
			|$best\_neighbor = find\_best\_neighbor( G )$|

\end{lstlisting}
