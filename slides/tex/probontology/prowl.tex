\subsection{PR-OWL}

\begin{frame}
	\begin{itemize}
		\item Extension of OWL used for Ontologies
		\item Used to represent probabilistic ontologies
		\item Based on MEBN theory
		\item Still an active area for research
	\end{itemize}
\end{frame}
